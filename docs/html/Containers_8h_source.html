253  std::array<double, 9> getStepRotationMatrix(unsigned n = 0) {
254  double Omega = h_twist * DEG_TO_RAD;
255  std::array<double, 9> m_mat {cos(Omega), -sin(Omega), 0, sin(Omega), cos(Omega), 0, 0, 0, 1};

274  std::array<double, 9> rodrigues_formula(std::array<double, 3> axis, double theta) {
275  std::array<double, 9> m{};
276  m[0] = cos(theta) + axis[0]*axis[0]*(1 - cos(theta));
277  m[1] = axis[0]*axis[1]*(1 - cos(theta)) - axis[2]*sin(theta);
278  m[2] = axis[1]*sin(theta) + axis[0]*axis[2]*(1 - cos(theta));
279  m[3] = axis[2]*sin(theta) + axis[0]*axis[1]*(1 - cos(theta));
280  m[4] = cos(theta) + axis[1]*axis[1]*(1 - cos(theta));
281  m[5] = -axis[0]*sin(theta) + axis[1]*axis[2]*(1 - cos(theta));
282  m[6] = -axis[1]*sin(theta) + axis[0]*axis[2]*(1 - cos(theta));
283  m[7] = axis[0]*sin(theta) + axis[1]*axis[2]*(1 - cos(theta));
284  m[8] = cos(theta) + axis[2]*axis[2]*(1 - cos(theta));
285 
286  return m;
287  };
