270  std::array<double, 9> rodrigues_formula(std::array<double, 3> axis, double theta) {
271  std::array<double, 9> m{};
272  m[0] = cos(theta) + axis[0]*axis[0]*(1 - cos(theta));
273  m[1] = axis[0]*axis[1]*(1 - cos(theta)) - axis[2]*sin(theta);
274  m[2] = axis[1]*sin(theta) + axis[0]*axis[2]*(1 - cos(theta));
275  m[3] = axis[2]*sin(theta) + axis[0]*axis[1]*(1 - cos(theta));
276  m[4] = cos(theta) + axis[1]*axis[1]*(1 - cos(theta));
277  m[5] = -axis[0]*sin(theta) + axis[1]*axis[2]*(1 - cos(theta));
278  m[6] = -axis[1]*sin(theta) + axis[0]*axis[2]*(1 - cos(theta));
279  m[7] = axis[0]*sin(theta) + axis[1]*axis[2]*(1 - cos(theta));
280  m[8] = cos(theta) + axis[2]*axis[2]*(1 - cos(theta));
281 
282  return m;
283  };
